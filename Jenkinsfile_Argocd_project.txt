pipeline {
    agent any

    tools {
        maven 'maven_3.9.16'
    }

    stages {

        stage('Checkout') {
            steps {
                echo 'Checkout Git repo'
                git branch: 'main',
                    url: 'https://github.com/aanirudhatom2294-lab/mindcircuit17d.git'
            }
        }

        stage('Build & Unit Test') {
            steps {
                sh 'mvn clean test'
            }
        }

        // stage('SonarQube Scan') {
        //     steps {
        //         echo 'Scanning project'
        //         sh 'ls -ltr'

        //         sh '''
        //             mvn sonar:sonar \
        //             -Dsonar.host.url=http://52.55.221.218:9000 \
        //             -Dsonar.token=squ_0f39becdb864b01079c7ca1c09c68e6d1353446f
        //         '''
        //     }
        // }

        stage('SonarQube Scanning') {
            steps {
                withSonarQubeEnv('SonarQube'){

                    sh '''
                        mvn clean sonar:sonar
                    '''
                }
            }
        }

        stage('Quality Gate Check') {
            steps {
                timeout(time: 5, unit: 'MINUTES') {
                    waitForQualityGate abortPipeline: true
                }
            }
        }

        stage('Building Artifact') {
            steps {
                echo 'Build Artifact'
                sh 'mvn clean package'
            }
        }

        stage('Building Docker Image') {
            steps {
                echo 'Build Docker Image'

                sh '''
                    docker build -t anirudhreddyk/argocd-project:${BUILD_NUMBER} -f Dockerfile .
                '''
            }
        }

        stage('Scan Docker Image using Trivy scan') {
            steps {
                echo 'Scanning Image'

                sh '''
                    trivy image anirudhreddyk/argocd-project:${BUILD_NUMBER}
                '''
            }
        }

        stage('Push to Docker Hub') {
            steps {
                script {
                    withCredentials([string(credentialsId: 'dockerhub', variable: 'dockerhub')]) {

                        sh '''
                            docker login -u devopshubg333 -p ${dockerhub}
                        '''

                        sh '''
                            docker push anirudhreddyk/argocd-project:${BUILD_NUMBER}
                        '''
                    }

                    echo 'Pushed to Docker Hub'
                }
            }
        }

        stage('Update Deployment File') {

            environment {
                GIT_REPO_NAME = "mindcircuit17d"
                GIT_USER_NAME = "aanirudhatom2294-lab"
            }

            steps {
                echo 'Update Deployment File'

                withCredentials([string(credentialsId: 'githubtoken', variable: 'githubtoken')]) {

                    sh '''
                        git config user.email "aanirudh.atom2294@gmail.com"
                        git config user.name "Anirudh"

                        sed -i "s/batch17d:.*/batch17d:${BUILD_NUMBER}/g" deploymentfiles/deploy.yaml

                        git add .
                        git commit -m "Update deployment image to version ${BUILD_NUMBER}" || true

                        git push https://${githubtoken}@github.com/${GIT_USER_NAME}/${GIT_REPO_NAME}.git HEAD:main
                    '''
                }
            }
        }
    }
}
